    def wrapper(*args, **kwargs):
        wrapper.count = wrapper.count + 1
        print("method: %s, count: %s" % (func.func_name, wrapper.count))
        return func(*args, **kwargs)

from functools import wraps

def localcache(func):
    cache = {}
    miss = object()
    
    @wraps(func)
    def wrapper(*args):
        result = cached.get(args, miss)
        if result is miss:
            result = func(*args)
            cached[args] = result
        return result
    
    return wrapper

class Router(object):
    
    def __init__(self):
        self.url_map = {}
        
    def register(self, url):
        def wrapper(func):
            self.url_map[url] = func

        return wrapper
        
    def call(self, url):
        func = self.url_map.get(url)
        if not func:
            raise ValueError("No url function: %s", url)
        
        return func()
    
router = Router()

@router.register("/page1")
def page1():
    return "This is page 1"

@router.register("/page2")
def page2():
    return "This is page 2"

print(router.call("/page1"))
print(router.call("/page2"))
